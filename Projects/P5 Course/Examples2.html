        createCanvas(400,400);
      }
      function draw(){
        noStroke();
        fill(120);
        //head
        ellipse(200, 30, 50, 50);
        rect(195,50,10,20);
        //belly
        rect(180, 70, 40, 100);
        //legs
        triangle(200, 150, 220, 150, 250, 250);
        triangle(180, 150, 200, 150, 150, 250);
        //arms
        triangle(200, 70, 220, 70, 250, 150);
        triangle(180, 70, 200, 70, 150, 150);      }

Example 6

    function setup() {
      createCanvas(400,300);
      background(0);
    }


    function draw() {
      stroke(255);
      strokeWeight(1);
      noFill();

      ellipse(mouseX + 10,mouseY, 10,10);
      ellipse(mouseX - 10,mouseY, 10,10);
      ellipse(mouseX,mouseY + 10, 10,10);
      ellipse(mouseX,mouseY -10, 10,10);
      };
    };
